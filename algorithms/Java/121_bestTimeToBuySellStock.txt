class Solution {
    public int maxProfit(int[] prices) {
    	// brute force
        // slide through and take the first element and compare it subtracted from the rest
        int maxprofit = 0;

        for (int i = 0; i < prices.length; i++) {
            int buy = prices[i];
            for (int j = i + 1; j < prices.length; j++) {
                int sell = prices[j];
                maxprofit = Math.max(maxprofit, sell - buy);
            }
        }
        return maxprofit;
    }
}
