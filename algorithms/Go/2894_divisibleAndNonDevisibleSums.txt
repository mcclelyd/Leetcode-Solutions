func differenceOfSums(n int, m int) int {
    var sum1 int = 0
    var sum2 int = 0
    for i := 1; i <= n; i++ {
        if (i % m != 0) {
            sum1 += i
        }
    }

    for j := 1; j <= n; j++ {
        if (j % m == 0) {
            sum2 += j
        }
    }

    return sum1 - sum2;
}
